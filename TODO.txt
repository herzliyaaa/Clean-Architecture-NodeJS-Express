app.use('/rct-api/docs', (req, res, next) => {
  const currentHost = `${req.protocol}://${req.hostname}:${port}`;

  const swaggerJson = require('../swagger-docs.json');
  swaggerJson.servers = [
    {
      url: `${currentHost}/rct-api`,
      description: "Feature Deployment",
    },
  ];

  req.swaggerDoc = swaggerJson;
  next();
}, swaggerUi.serve, (req, res) => {

  swaggerUi.setup(req.swaggerDoc)(req, res);
});
